<script>
function getDom(value){
    var selectDom = [];
    var dom=document.getElementsByClassName("hidden md:block")[0].getElementsByTagName('button');
    for (var i=0; i<dom.length; i++) {
        if(value===dom[i].innerText){
            selectDom.push(dom[i]);
        }
    }
    return selectDom;
}
getDom('默认')[0].click()
